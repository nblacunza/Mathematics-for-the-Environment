2 Coding in Julia

Core implementation of the Secant algorithm:

# Secant Method Implementation
function secant_method(f, x0, x1; tol=1e-8, max_iter=100)
    for i in 1:max_iter
        fx0, fx1 = f(x0), f(x1)
        # Avoid division by zero
        x2 = x1 - fx1 * (x1 - x0) / (fx1 - fx0)
        
        if abs(x2 - x1) < tol
            return x2 # Root found
        end
        x0, x1 = x1, x2
    end
    error("Method did not converge after $max_iter iterations")
end
